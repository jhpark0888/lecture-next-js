## 📁 폴더 구조

```
root/
├── assets/             # 정적 파일 (이미지, 폰트 등)
├── app/                # 화면 파일
│   ├── community/      # 커뮤니티 화면(미구현)
│   ├── meals/          # 음식 목록 화면
│   │   ├── share/      # 음식 추가 화면
│   │   ├── mealSlug/   # 음식 상세 화면
├── components/               
│   ├── images/         # 이미지 슬라이더 파일
│   ├── main-header/    # 헤더 파일
│   ├── meals/          # 음식 관련 컴포넌트 및 이미지 피커 
├── .gitignore          # Git 무시 설정 파일
├── package.json        # 프로젝트 설정 및 의존성
└── README.md           # 프로젝트 소개 문서
```
